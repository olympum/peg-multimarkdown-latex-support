% Change chapter style to default hangnum style of Memoir (see page 86 of the Memoir manual for the source), but with modifications to color, spacing and font size
\makechapterstyle{hangnum}{%
  \setlength{\afterchapskip}{20pt} % Modification: decrease default vertical space after chapter
  \renewcommand*{\chapnumfont}{\chaptitlefont}
  \settowidth{\chapindent}{\chapnumfont 999}
  \renewcommand*{\chapnamefont}{\normalfont\LARGE\sffamily\color{AppleChapter}} % Modification: decrease default font size
  \renewcommand*{\chaptitlefont}{\normalfont\LARGE\sffamily\color{AppleChapter}} % Modification: decrease default font size
  \renewcommand*{\printchaptername}{}
  \renewcommand*{\chapternamenum}{}
  \renewcommand*{\chapnumfont}{\chaptitlefont}
  \renewcommand*{\printchapternum}{%
    \noindent\llap{\makebox[\chapindent][l]{%
    \chapnumfont \thechapter}}}
  \renewcommand*{\afterchapternum}{}
}
\chapterstyle{hangnum} % Activate chapter style again for changes to take effect (not sure if it’s necessary for chapter styles too?)
